    １．查看编码：show variables like 'char%'；
    数据库表中字段的字符集设置 。show create table TableName
            set character_set_client = utf8;改变编码

   ２．成功解决了的解决办法
   linux 下更改配置文件，在/etc/mysql/my.ini或者my.cnf

创建用户表：
create database Wchat;
use Wchat;
create table user(
userid varchar(20) primary key,
username varchar(20) not null,
password varchar(20) not null,
age varchar(3),
sex varchar(4) not null default "男",
birthday date,
home varchar(40),
picture varchar(20),
usersession varchar(24)
);
